001  (ns file-server.server
002    (:require [clojure.java.io :refer [file]]
003              [file-server.server.handler :refer [app]]
004              [file-server.utils.io :refer [get-free-port]]
005              [ring.adapter.jetty :refer [run-jetty]])
006    (:gen-class))
007  
008  (defn start
009    "Takes first argument `port` to start the server on
010     And a second argument `serving-dir` which is the directory while will be served
011     If `port` is not specified, it will start the server on a random free port using `file-server.utils.io/get-free-port`
012     If `serving-dir` is not specified, it will start serving the current directory
013     NOTE: Both the arguments are optional"
014    ([] (start (get-free-port)))
015    ([port] (start port "."))
016    ([port serving-dir]
017     (if (.exists (file serving-dir))
018       (do (run-jetty (app serving-dir)
019                      {:join? false
020                       :port (cond-> port
021                               (string? port) Integer/parseInt)})
022           (println "Server started on port" port
023                    "and serving files under" (if (= serving-dir ".")
024                                                "current directory"
025                                                serving-dir)))
026       (throw (Exception. (str "`" serving-dir "` does not exist"))))))
027  
028  (defn -main [& args]
029    (apply start args))
